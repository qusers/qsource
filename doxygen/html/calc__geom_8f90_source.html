219 subroutine angle_calc(i)

221  integer, intent(in) :: i
222

225  real(8) :: a, v

249 subroutine torsion_calc(i)
250  !arguments
251  integer, intent(in) :: i
252 
253  !locals
254  real(8) :: rji(3), rjk(3), rkl(3), rnj(3), rnk(3), scp
255  real(8) :: phi, sgn, arg, v
256  integer :: ic
257 
258  if(i < 1 .or. i > nmeas) return
259 
260  rji(:) = xin(3*geom(i)%i-2:3*geom(i)%i) - xin(3*geom(i)%j-2:3*geom(i)%j)
261  rjk(:) = xin(3*geom(i)%k-2:3*geom(i)%k) - xin(3*geom(i)%j-2:3*geom(i)%j)
262  rkl(:) = xin(3*geom(i)%l-2:3*geom(i)%l) - xin(3*geom(i)%k-2:3*geom(i)%k)
263  !cross products
264  rnj(1) = rji(2) * rjk(3) - rji(3) * rjk(2)
265  rnj(2) = rji(3) * rjk(1) - rji(1) * rjk(3)
266  rnj(3) = rji(1) * rjk(2) - rji(2) * rjk(1)
267  rnk(1) = - rjk(2) * rkl(3) + rjk(3) * rkl(2)
268  rnk(2) = - rjk(3) * rkl(1) + rjk(1) * rkl(3)
269  rnk(3) = - rjk(1) * rkl(2) + rjk(2) * rkl(1)
270  scp = dot_product(rnj, rnk) &
271  / sqrt(dot_product(rnj, rnj)*dot_product(rnk, rnk))
272  IF(scp>1.0) scp = 1.0
273  IF(scp< - 1.0) scp = - 1.0
274  phi = acos(scp)
275  sgn = rjk(1) *(rnj(2) * rnk(3) - rnj(3) * rnk(2) ) + rjk(2)&
276  *(rnj(3) * rnk(1) - rnj(1) * rnk(3) ) + rjk(3) *(rnj(1) &
277  * rnk(2) - rnj(2) * rnk(1) )
278  IF(sgn<0) phi = - phi
279  write(*,100, advance='no') phi*180/pi
280 100 format(f10.2)
281 
282  ic = geom(i)%cod
283  if(ic > 0) then !calc energy
284  arg = torlib(ic)%rmult * phi - torlib(ic)%deltor * pi / 180.
285  v = torlib(ic)%fk *(1.0 + cos(arg) ) / real(torlib(ic)%paths )
286  write(*,110, advance='no') v
287  end if
288 110 format(f8.2)

290 end subroutine torsion_calc
