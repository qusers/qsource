382 subroutine schlittersformula(t, det)
383 
384  type(trajectory) :: t

386  real(8) :: det

388  call multiplymatrices(t) ! MCM + 1

390

392  det = 0.0
393  do i=1,t%no_atoms*3
394  det = det + 2*log(t%C(i,i)) ! det L*L = 2 * det L
395  end do

405 subroutine multiplymatrices(t)
406 
407  type(trajectory) :: t
408  integer :: i,j,k,l,n
409  real(8) :: det, c !, kb, hs, e, c, u
410 
411  ! Set constants
412 
413  ! kb=1.38066 ! Boltzmann konstant *10**(-23)
414  ! hs=1.0545727 ! Plancs constant *10**(-34)
415  ! e=2.718281828 ! Euler number
416  ! u=1.6605402 ! Atomic mass constant *10**(-27) from u to Kg
417 
418  c=(e**2)*kb*(t%Temperature)*u*0.01/(hs**2) ! 0.01 is to m, u is u to kg (hs*hs) to hs**2
419 
420  do i=1,t%no_atoms*3
421  do j=1,i !t%no_atoms*3
422 
423  ! Multiply with constant
424  t%C(i,j) = c*t%C(i,j)*sqrt(t%massvector(i)*t%massvector(j))
425  t%C(j,i) = t%C(i,j) ! Symmetrix matrix
426 
427  ! Add diagonal matrix to massweigted matrix
428  if( i == j ) then
429  t%C(i,i) = t%C(i,i) + 1.0
430  endif
431 
432  enddo
433  enddo
434 
435 end subroutine multiplymatrices

437 !#############################################
438 !- Choleskyfactorization of symmetric matrix -
439 !#############################################
440 
441 ! Coleskyfactorizes the covariance matrix and
442 ! and stores the result (L) in C, C = L*L
443 ! Heath, M., Scientific Computing - An Introduction Survey
444 
445 subroutine choleskyfactorization(t)
446 
447  type(trajectory) :: t
448  integer :: i,j,k,l
449 
450  do j = 1,t%no_atoms*3
451  do k = 1,j-1
452  do i = j,t%no_atoms*3
453  t%C(i,j) = t%C(i,j) - t%C(i,k)*t%C(j,k)
454  enddo
455  enddo
456  if( t%C(j,j) < 0.0 ) then
457  print*, "Matrix is not positive definite!" ! Should hold.
458  endif
459  t%C(j,j) = sqrt(t%C(j,j))
460  do k=j+1,t%no_atoms*3
461  t%C(k,j) = t%C(k,j)/t%C(j,j)
462  end do
463  end do
464 
465 end subroutine choleskyfactorization
